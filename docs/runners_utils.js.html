/**
 * Utilities for running tournaments.
 * @module lib/runners/utils
 */

var _ = require('underscore')

/**
 * Determines the offset (between +2 and -2) for `ROUND_LENGTH`
 * so it will remain unknown to strategies prior to runtime.
 * @const
 * @type {Number}
 */
const ROUND_LENGTH_VARIANCE = Math.floor(Math.random() * 5) - 2

/**
 * The number of turns in each round. Sometimes 100 but not always.
 * @const
 * @type {Number}
 */
const ROUND_LENGTH = 100 + ROUND_LENGTH_VARIANCE

/**
 * Tallies each player's score from their actions during the round.
 * Values are derived from Poundstone's 1992 paper.
 * Assumes that the length of both histories are equal to
 * `ROUND_LENGTH`.
 * @param  {Array} history1 An array of the first player's actions
 * @param  {Array} history2 An array of the second player's actions.
 * @return {Array}          Each player's score during the round, as `[x, y]`.
 */
function score (history) {
  var scores = [0, 0]
  var history1 = history[0]
  var history2 = history[1]
  for (var i = 0; i < ROUND_LENGTH; i++) {
    // T T -- 1 1
    if (history1[i] && history2[i]) {
      scores[0]++
      scores[1]++
    } else if (history1[i] || history2[i]) {
      // F T -- 0 3
      // T F -- 3 0
      if (history1[i]) {
        scores[0] += 3
      } else {
        scores[1] += 3
      }
    } else {
    // F F -- 2 2
      scores[0] += 2
      scores[1] += 2
    }
  }

  return scores
}

/**
 * Pit two strategies against each other, returning the score
 * of each strategy.
 * @param  {Function} s1 A strategy function.
 * @param  {Function} s2 Another strategy function.
 * @return {Array}       Each player's score during the round, as `[x, y]`.
 */
function play (strat1, strat2) {
  var strat1History = []
  var strat2History = []

  for (var i = 0; i < ROUND_LENGTH; i++) {
    var strat1Choice = strat1(_.zip(strat1History, strat2History))
    var strat2Choice = strat2(_.zip(strat2History, strat1History))
    strat1History.push(strat1Choice)
    strat2History.push(strat2Choice)
  }

  return [strat1History, strat2History]
}

exports.ROUND_LENGTH = ROUND_LENGTH
exports.play = play
exports.score = score
